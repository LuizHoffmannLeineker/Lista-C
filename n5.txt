struct no{
    int valor;
    struct no *proximo;
    int indice;
};

void buscarElemento(struct no *atual, int key){
    int aux = 0;
    while(atual != NULL){
        if(atual->valor == key){
            printf("\nelemento foi encontrado no %d", atual->indice);
            aux = 1;
        }
        atual = atual->proximo;
    }
    
    if(aux != 1){
        printf("\no elemento nÃ£o foi encontrado!\n");
    }
    
}

void imprimirLista(struct no *atual){
    while(atual != NULL){
        printf("%d ", atual->valor);
        atual = atual->proximo;
    }
}
